```bash

  -b /home/work/Desktop/variants/ASSEMBLY_TRUTH_SET_MINMAP_FINAL/variants_minimap2.normalized.vcf.gz \
  -c /home/work/Desktop/variants/new_latest_all_data/snvs/working/snps/BCF_normal_SNPs.vcf.gz \
